        function calculatePremium() {

            const topup = document.getElementById("topup").value;

            const joiningDateValue = document.getElementById("joiningDate").value;

            const joiningDate = new Date(joiningDateValue);

            const currentDate = new Date();

            let totalPremium = 0;


            if (joiningDateValue && (currentDate - joiningDate) <= 365 * 24 * 60 * 60 * 1000) {

                const joiningMonth = joiningDate.getMonth() + 1; // getMonth() returns 0-11

                const remainingMonths = 12 - joiningMonth + 1;


                for (let i = 1; i <= memberCount; i++) {

                    const ageInput = document.getElementById(`age${i}`);

                    if (ageInput && ageInput.value) {

                        const age = parseInt(ageInput.value) || 0;

                        totalPremium += (getPremium(age, topup) / 12) * remainingMonths;

                    }

                }

            } else {

                for (let i = 1; i <= memberCount; i++) {

                    const ageInput = document.getElementById(`age${i}`);

                    if (ageInput && ageInput.value) {

                        const age = parseInt(ageInput.value) || 0;

                        totalPremium += getPremium(age, topup);

                    }

                }

            }


            document.getElementById("totalPremium").innerText = totalPremium.toFixed(2);

        }
